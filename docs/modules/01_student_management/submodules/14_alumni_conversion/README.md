### Rule 1: Automatic Alumni Conversion

```javascript
FUNCTION convert_to_alumni(student_id) {
  // Get student details
  student = GET_STUDENT(student_id)
  
  // Check if Grade 12 and passed
  IF student.current_grade != '12' {
    RETURN {success: FALSE, error: "Not a Grade 12 student"}
  }
  
  // Check if board results declared
  board_result = GET_BOARD_RESULT(student_id)
  IF NOT board_result OR board_result.status != 'PASSED' {
    RETURN {success: FALSE, error: "Board result not available or failed"}
  }
  
  // Generate alumni number
  alumni_number = GENERATE_ALUMNI_NUMBER(student.graduation_year)
  
  // Create alumni record
  INSERT INTO alumni (
    student_id,
    alumni_number,
    batch_year,
    graduation_date,
    final_grade,
    final_percentage,
    alumni_email
  ) VALUES (
    student_id,
    alumni_number,
    student.graduation_year,
    GET_ACADEMIC_YEAR_END(student.academic_year),
    student.current_grade,
    board_result.percentage,
    GENERATE_ALUMNI_EMAIL(student.first_name, student.last_name, alumni_number)
  )
  
  // Update student status
  UPDATE students SET
    student_status = 'ALUMNI',
    graduation_date = GET_ACADEMIC_YEAR_END(student.academic_year)
  WHERE student_id = student_id
  
  // Send welcome email
  SEND_ALUMNI_WELCOME_EMAIL(student_id, alumni_number)
  
  // Create alumni portal access
  CREATE_ALUMNI_PORTAL_ACCESS(student_id)
  
  RETURN {
    success: TRUE,
    alumni_number: alumni_number,
    alumni_email: alumni_email
  }
}
```
