  -- Foreign Keys
  FOREIGN KEY (student_id) REFERENCES students(student_id),
  
  -- Indexes
  INDEX idx_student_year (student_id, academic_year)

### Rule 1: Co-scholastic Grade Calculation

```javascript
FUNCTION calculate_cocholastic_grade(student_id, academic_year) {
  // Get all co-scholastic assessments
  life_skills = GET_LIFE_SKILLS_RATING(student_id, academic_year)
  pe_marks = GET_PE_MARKS(student_id, academic_year)
  arts_marks = GET_ARTS_MARKS(student_id, academic_year)
  value_ed_marks = GET_VALUE_ED_MARKS(student_id, academic_year)
  
  // Calculate average
  total = (life_skills + pe_marks + arts_marks + value_ed_marks) / 4
  
  // Convert to grade (CBSE 5-point scale)
  IF total >= 90 THEN grade = 'A'
  ELSE IF total >= 75 THEN grade = 'B'
  ELSE IF total >= 60 THEN grade = 'C'
  ELSE IF total >= 45 THEN grade = 'D'
  ELSE grade = 'E'
  
  RETURN grade
}
```
