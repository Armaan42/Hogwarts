## üìä DATABASE SCHEMA

### Board Compliance Table

```sql
CREATE TABLE board_compliance (
  compliance_id INT PRIMARY KEY AUTO_INCREMENT,
  
  -- Board Details
  board ENUM('CBSE', 'ICSE', 'IB_PYP', 'IB_MYP', 'IB_DP', 'CAMBRIDGE_PRIMARY', 'CAMBRIDGE_IGCSE', 'CAMBRIDGE_A_LEVEL', 'STATE') NOT NULL,
  
  -- Academic Year
  academic_year VARCHAR(9) NOT NULL,
  
  -- Compliance Checklist
  compliance_checklist JSON NOT NULL,
  compliance_percentage DECIMAL(5,2),
  
  -- Audit
  last_audit_date DATE,
  next_audit_date DATE,
  audit_status ENUM('PENDING', 'SCHEDULED', 'COMPLETED', 'DEFICIENCIES_FOUND') DEFAULT 'PENDING',
  
  -- Deficiencies
  deficiencies JSON,
  action_plan TEXT,
  action_plan_deadline DATE,
  
  -- Affiliation
  affiliation_number VARCHAR(100),
  affiliation_valid_until DATE,
  affiliation_status ENUM('ACTIVE', 'RENEWAL_PENDING', 'EXPIRED') DEFAULT 'ACTIVE',
  
  -- Metadata
  created_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  modified_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  
  -- Indexes
  INDEX idx_board_year (board, academic_year),
  INDEX idx_audit_status (audit_status),
  INDEX idx_affiliation_status (affiliation_status)
);
```

### Rule 1: Compliance Monitoring & Alerts

```javascript
FUNCTION monitor_board_compliance(board, academic_year) {
  compliance = GET_COMPLIANCE(board, academic_year)
  
  // Check compliance percentage
  IF compliance.compliance_percentage < 90 {
    SEND_ALERT("academic_coordinator", {
      title: "Board Compliance Below Target",
      message: `${board} compliance at ${compliance.compliance_percentage}% (Target: 100%)`,
      priority: "HIGH"
    })
  }
  
  // Check affiliation expiry
  days_to_expiry = compliance.affiliation_valid_until - CURRENT_DATE()
  
  IF days_to_expiry < 90 {
    SEND_ALERT("principal", {
      title: "Affiliation Renewal Required",
      message: `${board} affiliation expires in ${days_to_expiry} days`,
      priority: "CRITICAL"
    })
  }
  
  // Check pending deficiencies
  IF compliance.deficiencies.length > 0 {
    overdue = compliance.action_plan_deadline < CURRENT_DATE()
    
    IF overdue {
      SEND_ALERT("principal", {
        title: "Compliance Deficiencies Overdue",
        message: `${compliance.deficiencies.length} deficiencies past deadline`,
        priority: "CRITICAL"
      })
    }
  }
  
  RETURN compliance
}
```
